Edit
Please press Shift + Enter 
@Component({
  selector: 'body',
  templateUrl: './app.html'
})
export class AppComponent {
  protected window = window;
  protected editText: string = 'Input text\nLine 2';
}
<ngx-emoji contenteditable="true"
    [text]="editText"
    (text)="editText = $event"
    [enterOn]="{shift: true, ctrl: false}"
    (enter)="window.alert(editText);"
></ngx-emoji>

Chat
Press Ctrl + Enter to send message 

@Component({
  selector: 'body',
  templateUrl: './app.html'
})
export class AppComponent {
  protected messages: string[] = [];
}
            

<ngx-emoji *ngFor="let message of messages;"
    [text]="message"
></ngx-emoji>
<ngx-emoji contenteditable="true"
    (text)="messages.push($event)"
    [enterOn]="{ctrl: true}"
    #chatInput
></ngx-emoji>
